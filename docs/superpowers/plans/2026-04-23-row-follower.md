**Goal:** Build a ROS 2 node that uses the OAK-D top-mounted camera to autonomously drive the Amiga robot down a single corn row in the `virtual_maize_field` Ignition Gazebo sim, stops when the row ends, and logs per-frame numeric data to CSV.

**Architecture:** One colcon package (`row_follower`), one rclpy node. Perception is stateless pure functions (unit-testable without ROS); the node owns state (mode, CSV writer, rolling buffers). Data flow: `/oak/rgb` → HSV mask → stacked-band centerline fit → P controller → `/cmd_vel`. Debug overlay published on `/row_follower/debug_image`.

**Tech Stack:** ROS 2 Humble, rclpy, cv_bridge, OpenCV, NumPy, pytest, ament_python.

Produces three 640x480 BGR PNGs:
  good_row.png     — two vertical green stripes straddling image center (robot centered)
  end_of_row.png   — stripes only in the bottom quarter (robot at row's end)
  empty.png        — uniform brown "soil" (no crops)

from pathlib import Path

import cv2
import numpy as np

FIXTURES = Path(__file__).parent / "fixtures"
WIDTH, HEIGHT = 640, 480
SOIL_BGR = (40, 60, 90)        # brownish
GREEN_BGR = (40, 180, 40)      # corn green


def _blank() -> np.ndarray:
    img = np.zeros((HEIGHT, WIDTH, 3), dtype=np.uint8)
    img[:] = SOIL_BGR
    return img


def good_row() -> np.ndarray:
    img = _blank()
    # Two green vertical stripes centered around x=320
    for cx in (220, 420):
        cv2.rectangle(img, (cx - 20, 0), (cx + 20, HEIGHT), GREEN_BGR, -1)
    return img


def end_of_row() -> np.ndarray:

    # Stripes only in the very bottom quarter — robot has passed the row
    for cx in (220, 420):
        cv2.rectangle(img, (cx - 20, int(HEIGHT * 0.85)), (cx + 20, HEIGHT), GREEN_BGR, -1)
    return img


def empty() -> np.ndarray:
    return _blank()


def main() -> None:
    FIXTURES.mkdir(parents=True, exist_ok=True)
    cv2.imwrite(str(FIXTURES / "good_row.png"), good_row())
    cv2.imwrite(str(FIXTURES / "end_of_row.png"), end_of_row())
    cv2.imwrite(str(FIXTURES / "empty.png"), empty())
    print(f"Wrote 3 fixtures to {FIXTURES}")


if __name__ == "__main__":
    main()

"""Pure-function crop-row perception — no ROS, no I/O, no state.

Everything in this module takes numpy arrays in and returns numpy/plain types
so it can be unit-tested against PNG fixtures.
"""
from __future__ import annotations

from dataclasses import dataclass

import cv2
import numpy as np


@dataclass(frozen=True)
class HSVThresholds:
    h_min: int
    h_max: int
    s_min: int
    s_max: int
    v_min: int
    v_max: int


def hsv_mask(bgr: np.ndarray, hsv: HSVThresholds) -> np.ndarray:
    """Return a uint8 {0, 255} mask of green pixels in `bgr`.

    BGR → HSV → inRange → open(3x3) → close(5x5) to denoise.
    """
    img_hsv = cv2.cvtColor(bgr, cv2.COLOR_BGR2HSV)
    lower = np.array([hsv.h_min, hsv.s_min, hsv.v_min], dtype=np.uint8)
    upper = np.array([hsv.h_max, hsv.s_max, hsv.v_max], dtype=np.uint8)
    mask = cv2.inRange(img_hsv, lower, upper)
    open_k = cv2.getStructuringElement(cv2.MORPH_RECT, (3, 3))
    close_k = cv2.getStructuringElement(cv2.MORPH_RECT, (5, 5))
    mask = cv2.morphologyEx(mask, cv2.MORPH_OPEN, open_k)
    mask = cv2.morphologyEx(mask, cv2.MORPH_CLOSE, close_k)
    return mask
```

def green_forward_count(mask: np.ndarray, roi_top_frac: float) -> int:
    """Count mask pixels inside the forward sub-ROI.

    The ROI is the bottom `(1 - roi_top_frac)` of the image. The forward
    sub-ROI is the upper half of that ROI (i.e. the part of the image the
    robot is driving *into*).
    """
    h, _w = mask.shape
    roi_top = int(h * roi_top_frac)
    roi_bottom = h
    roi_mid = (roi_top + roi_bottom) // 2
    forward = mask[roi_top:roi_mid, :]
    return int((forward > 0).sum())
```

def _band_midpoint(band_mask: np.ndarray, image_center_x: int) -> float | None:
    """Return x-coord of midpoint between the two peaks straddling center,
    or None if fewer than two usable peaks exist.
    """
    if band_mask.sum() == 0:
        return None
    col_sum = band_mask.sum(axis=0).astype(np.int32)
    # Smooth the histogram to make peak-finding robust to noise.
    kernel = np.ones(11, dtype=np.float32) / 11.0
    smoothed = np.convolve(col_sum, kernel, mode="same")
    left = smoothed[:image_center_x]
    right = smoothed[image_center_x:]
    if left.max() == 0 or right.max() == 0:
        return None
    left_peak = int(np.argmax(left))
    right_peak = int(np.argmax(right)) + image_center_x
    return (left_peak + right_peak) / 2.0


def find_centerline(
    mask: np.ndarray,
    n_bands: int,
    roi_top_frac: float,
) -> tuple[float, float] | None:
    """Fit a centerline through `n_bands` horizontal bands of the mask ROI.

    Returns (offset_px, heading_rad):
      offset_px  — x-distance of the fitted line at the bottom of the ROI
                   from image center (positive = line is right of center,
                   which means the robot should steer right... no — positive
                   means the midpoint between rows is right of the camera,
                   so the robot is LEFT of center and should steer right).
      heading_rad — atan2(dx, dy) of the fitted line; 0 means vertical.

    Returns None if fewer than 2 bands produce valid midpoints.
    """
    h, w = mask.shape
    roi_top = int(h * roi_top_frac)
    roi = mask[roi_top:, :]
    roi_h = roi.shape[0]
    if roi_h == 0 or n_bands <= 0:
        return None

    band_h = roi_h // n_bands
    if band_h == 0:
        return None

    image_center_x = w // 2
    midpoints: list[tuple[float, float]] = []  # (y_in_full_image, mid_x)
    for i in range(n_bands):
        y0 = i * band_h
        y1 = (i + 1) * band_h if i < n_bands - 1 else roi_h
        band = roi[y0:y1, :]
        mid_x = _band_midpoint(band, image_center_x)
        if mid_x is not None:
            y_center = roi_top + (y0 + y1) / 2.0
            midpoints.append((y_center, mid_x))

    if len(midpoints) < 2:
        return None

    ys = np.array([p[0] for p in midpoints], dtype=np.float64)
    xs = np.array([p[1] for p in midpoints], dtype=np.float64)
    # Fit x = m*y + b; slope m is dx/dy so heading = atan2(dx, dy) = atan(m)
    slope, intercept = np.polyfit(ys, xs, 1)
    x_at_bottom = slope * (h - 1) + intercept
    offset_px = float(x_at_bottom - image_center_x)
    heading_rad = float(np.arctan(slope))
    return offset_px, heading_rad
```

"""RowFollower ROS 2 node — subscribes to /oak/rgb and /odom, publishes /cmd_vel.

This is the scaffolding commit: wiring + a 10 Hz control timer that currently
just emits Twist(0). State machine, perception, logging, and debug publishing
are layered on in subsequent tasks.
"""
from __future__ import annotations

import rclpy
from geometry_msgs.msg import Twist
from nav_msgs.msg import Odometry
from rclpy.node import Node
from sensor_msgs.msg import Image


class RowFollower(Node):
    def __init__(self) -> None:
        super().__init__("row_follower")

        # Parameters are declared here so they show up in `ros2 param list`.
        # Values get used in later tasks.
        self.declare_parameter("hsv_h_min", 35)
        self.declare_parameter("hsv_h_max", 85)
        self.declare_parameter("hsv_s_min", 40)
        self.declare_parameter("hsv_s_max", 255)
        self.declare_parameter("hsv_v_min", 40)
        self.declare_parameter("hsv_v_max", 255)
        self.declare_parameter("roi_top_frac", 0.4)
        self.declare_parameter("n_bands", 4)
        self.declare_parameter("end_of_row_pixels", 500)
        self.declare_parameter("end_of_row_consecutive_frames", 10)
        self.declare_parameter("lost_detection_frames", 15)
        self.declare_parameter("v_nominal", 0.3)
        self.declare_parameter("k_lat", 1.2)
        self.declare_parameter("k_hdg", 1.5)
        self.declare_parameter("slowdown", 0.5)
        self.declare_parameter("csv_dir", "~/.ros/row_follower")

        self._latest_image: Image | None = None
        self._latest_odom: Odometry | None = None

        self._cmd_pub = self.create_publisher(Twist, "/cmd_vel", 10)
        self.create_subscription(Image, "/oak/rgb", self._on_image, 10)
        self.create_subscription(Odometry, "/odom", self._on_odom, 10)

        # 10 Hz control loop. Image callback stores the frame; the timer
        # processes whatever the latest frame is and emits a Twist.
        self.create_timer(0.1, self._on_tick)

        self.get_logger().info("RowFollower node started")

    def _on_image(self, msg: Image) -> None:
        self._latest_image = msg

    def _on_odom(self, msg: Odometry) -> None:
        self._latest_odom = msg

    def _on_tick(self) -> None:
        # Scaffold: always publish zero. Control logic added in Task 8.
        self._cmd_pub.publish(Twist())


def main(args: list[str] | None = None) -> None:
    rclpy.init(args=args)
    node = RowFollower()
    try:
        rclpy.spin(node)

        if self._latest_image is None:
            self._cmd_pub.publish(Twist())
            return

        bgr = self._bridge.imgmsg_to_cv2(self._latest_image, desired_encoding="bgr8")
        hsv = self._hsv_from_params()
        mask = perception.hsv_mask(bgr, hsv)
        roi_top_frac = float(self.get_parameter("roi_top_frac").value)
        n_bands = int(self.get_parameter("n_bands").value)
        detection = perception.find_centerline(mask, n_bands=n_bands, roi_top_frac=roi_top_frac)

        self._update_state(detection, green_fwd)

        self._cmd_pub.publish(Twist())

    # -- State machine ------------------------------------------------------

    def _update_state(
        self,
        detection: tuple[float, float] | None,
        green_fwd: int,
    ) -> None:
        eor_threshold = int(self.get_parameter("end_of_row_pixels").value)
        lost_limit = int(self.get_parameter("lost_detection_frames").value)

        self._eor_buffer.append(green_fwd)
        buffer_full = len(self._eor_buffer) == self._eor_buffer.maxlen
        all_below = buffer_full and all(c < eor_threshold for c in self._eor_buffer)

        if self.state == "STOPPED":
            return
        if self.state == "WAITING":
            if detection is not None:

            self.get_logger().info("FOLLOWING -> STOPPED (end of row)")
            return

            if self._lost_frames >= lost_limit:

                self.get_logger().info("FOLLOWING -> STOPPED (lost detection)")
        else:
            self._lost_frames = 0

            v_max=int(g("hsv_v_max").value),
        )

    def _open_csv(self) -> None:
        if self._csv_file is not None:
            try:
                self._csv_file.close()
            except Exception:  # noqa: BLE001
                pass
            self._csv_file = None
            self._csv_writer = None

        csv_dir = os.path.expanduser(str(self.get_parameter("csv_dir").value))
        try:
            Path(csv_dir).mkdir(parents=True, exist_ok=True)
            ts = datetime.now().strftime("%Y%m%d_%H%M%S")
            path = Path(csv_dir) / f"run_{ts}.csv"
            self._csv_file = open(path, "w", newline="")
            self._csv_writer = csv.writer(self._csv_file)
            self._csv_writer.writerow([
                "t_sec", "state", "odom_x", "odom_y", "odom_yaw",
                "offset_px", "heading_rad", "green_px_forward", "cmd_v", "cmd_w",
            ])
            self._csv_file.flush()
            self.get_logger().info(f"Logging to {path}")
        except OSError as exc:  # disk/permission — do not stop the robot
            self.get_logger().error(f"Cannot open CSV: {exc}")
            self._csv_file = None
            self._csv_writer = None

    def _log_row(
        self,
        detection: tuple[float, float] | None,
        green_fwd: int,
        cmd: Twist,
    ) -> None:
        if self._csv_writer is None:
            return
        now = self.get_clock().now().nanoseconds / 1e9
        t_sec = now - self._start_time
        if self._latest_odom is None:
            ox = oy = oyaw = ""
        else:
            p = self._latest_odom.pose.pose
            ox = p.position.x
            oy = p.position.y
            # Yaw from quaternion (z-axis rotation only)
            q = p.orientation
            oyaw = math.atan2(2.0 * (q.w * q.z + q.x * q.y),
                              1.0 - 2.0 * (q.y * q.y + q.z * q.z))
        off = "" if detection is None else detection[0]
        hdg = "" if detection is None else detection[1]
        try:
            self._csv_writer.writerow([
                f"{t_sec:.3f}", self.state, ox, oy, oyaw,
                off, hdg, green_fwd, cmd.linear.x, cmd.angular.z,
            ])
            self._csv_file.flush()
        except OSError as exc:  # noqa: BLE001
            self.get_logger().error(f"CSV write failed: {exc}")

                    cv2.circle(out, (int(mid_x), y_center), 5, (0, 0, 255), -1)
            # Fitted line: reconstruct from offset_px and heading_rad.
            slope = math.tan(heading_rad)
            x_at_bottom = w / 2.0 + offset_px
            x_at_top = x_at_bottom - slope * (h - 1 - roi_top)
            cv2.line(
                out,
                (int(x_at_top), roi_top),
                (int(x_at_bottom), h - 1),
                (0, 255, 255),
                2,
            )

        # Status text.
        status = f"{self.state}  off={0 if detection is None else int(detection[0])}px  g_fwd={green_fwd}"
        cv2.putText(out, status, (10, 25), cv2.FONT_HERSHEY_SIMPLEX, 0.7, (255, 255, 255), 2)
        return out
```
